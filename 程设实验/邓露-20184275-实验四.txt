#include <iostream>
using namespace std;

class Shape
{
public:
	Shape(){}
	~Shape(){}
	virtual float GetArea()
	{
		return 0;
	}
};
class Circle : public Shape
{
public:
	Circle(float r):radius(r){}
	~Circle(){};
	float GetArea()
	{
		return 3.14f*radius*radius;
	}
private:
	float radius;

};

class Rectangle : public Shape
{
private:
	float m_height,m_width;
public:
	Rectangle(float height, float width):m_height(height),m_width(width){}
	~Rectangle(){}
	float GetArea()
	{
		return m_height*m_width;
	}
	float GetHeight()
	{
		return m_height;
	}
	float GetWidth()
	{
		return m_width;
	}
};

int main()
{
	Shape*sp;
	sp=new Circle(6);
	cout<<"The area of the circle is "<<sp->GetArea()<<endl;
	delete sp;
	sp=new Rectangle(3,4);
	cout<<"The area of the rectangle is "<<sp->GetArea()<<endl;
	delete sp;
}

#include <iostream>
#include<string>

using namespace std;
class tank
{
public:
	string color;
	string thickness;
	string speed;
	int num;
	tank()
	{
		color="0";
		thickness="0";
		speed="0";
		num=0;
	}
	void Show()
	{
		cout<<"The color is: "<<this->color<<endl;
		cout<<"The thickness is: "<<this->thickness<<endl;
		cout<<"The speed is: "<<this->speed<<endl;
		cout<<"The number is: "<<this->num<<endl;
	}
	virtual ~tank();
};

class supertank : public tank
{
public:
	supertank()
	{
		num=1;
		speed="slow";
		color="green";
		thickness="thick";
	}
};

class tankette : public tank
{
public:
	tankette()
	{
		num=2;
		speed="fast";
		color="gray";
		thickness="thin";
	}
};

int main()
{
	cout<<"the quality of the supertank: "<<endl;
	supertank a;
	a.Show();
	cout<<"the quality of the tankette: "<<endl;
	tankette b;
	b.Show();
	return 0;

}
